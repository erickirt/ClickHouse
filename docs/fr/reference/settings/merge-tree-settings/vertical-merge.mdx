  ## vertical_merge_optimize_ttl_delete
</div>

<SettingsInfoBlock type="Bool" default_value="1" />

<VersionHistory rows={[{"id": "row-1","items": [{"label": "26.3"},{"label": "1"},{"label": "Autoriser l’algorithme de fusion verticale pour les fusions devant supprimer les lignes expirées par TTL"}]}]} />
